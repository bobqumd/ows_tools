<script>

    // Set Chart.js Dark Mode Defaults

    Chart.defaults.color = '#9CA3AF';

    Chart.defaults.font.family = "'Segoe UI', sans-serif";


    const dieTypes = [4, 6, 8, 10, 12, 16, 20];

    const attackInputs = document.getElementById('attack-inputs');

    

    // Build Input Fields

    for(let i=1; i<=4; i++) {

        let defaultDie = i===1 ? 12 : (i===2 ? 10 : (i===3 ? 8 : 20));

        let defaultCount = i===1 ? 1 : 0; 

        let minCount = i===1 ? 1 : 0;

        

        let html = `<div class="flex-row">

            <div>

                <label style="font-size:0.7rem; color:#6B7280;">Attack Value ${i}</label><br>

                <select id="die${i}" onchange="calculate()" style="width:100%; margin-top:5px;">

                    ${dieTypes.map(d => `<option value="${d}" ${d===defaultDie ? 'selected':''}>D${d}</option>`).join('')}

                </select>

            </div>

            <div>

                <label style="font-size:0.7rem; color:#6B7280;">Salvoes ${i}</label><br>

                <input type="number" id="count${i}" value="${defaultCount}" min="${minCount}" max="24" onchange="calculate()" style="width:90%; margin-top:5px;">

            </div>

        </div>`;

        attackInputs.innerHTML += html;

    }


    let myChart = null;


    function getProb(sides, target) {

        return target > sides ? 0 : (sides - target + 1) / sides;

    }


    function calcPkForPackage(packageArray, red, grayCount, grayVal, steps) {

        let totalSalvos = packageArray.length;

        if(totalSalvos === 0) return { pk: 0, pmf: [1] };


        packageArray.sort((a,b) => b-a);

        let excess = Math.max(0, grayCount - totalSalvos);

        let demotions = new Array(totalSalvos).fill(0);

        

        for(let i=0; i<excess; i++) {

            demotions[i % totalSalvos]++;

        }


        let probs = [];

        for(let i=0; i<totalSalvos; i++) {

            let baseSides = packageArray[i];

            let dCount = demotions[i];

            let idx = dieTypes.indexOf(baseSides);

            let effSides = dieTypes[Math.max(0, idx - dCount)];

            

            let pGray = grayCount > 0 ? getProb(effSides, grayVal) : 1;

            let pRed = getProb(effSides, red);

            probs.push(pGray * pRed);

        }


        let pmf = [1.0];

        for(let p of probs) {

            let nextPmf = new Array(pmf.length + 1).fill(0);

            for(let j=0; j<pmf.length; j++) {

                nextPmf[j] += pmf[j] * (1-p);

                nextPmf[j+1] += pmf[j] * p;

            }

            pmf = nextPmf;

        }


        let pk = 0;

        for(let i=steps; i<pmf.length; i++) pk += pmf[i];

        

        return { pk, pmf };

    }


    function calculate() {

        let c1 = parseInt(document.getElementById('count1').value) || 0;

        let c2 = parseInt(document.getElementById('count2').value) || 0;

        let c3 = parseInt(document.getElementById('count3').value) || 0;

        let c4 = parseInt(document.getElementById('count4').value) || 0;

        

        let d1 = parseInt(document.getElementById('die1').value);

        let d2 = parseInt(document.getElementById('die2').value);

        let d3 = parseInt(document.getElementById('die3').value);

        let d4 = parseInt(document.getElementById('die4').value);


        let red = parseInt(document.getElementById('redShield').value);

        let steps = parseInt(document.getElementById('targetSteps').value);

        let grayCount = parseInt(document.getElementById('grayShields').value);

        let grayVal = parseInt(document.getElementById('grayValue').value);


        let isSingleDie = (c2 === 0 && c3 === 0 && c4 === 0);

        

        let packageArray = [];

        for(let i=0; i<c1; i++) packageArray.push(d1);

        for(let i=0; i<c2; i++) packageArray.push(d2);

        for(let i=0; i<c3; i++) packageArray.push(d3);

        for(let i=0; i<c4; i++) packageArray.push(d4);


        let currentRes = calcPkForPackage(packageArray, red, grayCount, grayVal, steps);

        document.getElementById('pkResult').innerText = (currentRes.pk * 100).toFixed(1) + '%';


        if(myChart) myChart.destroy();

        const ctx = document.getElementById('resultChart').getContext('2d');

        

        // Chart Grid Configuration for Dark Mode

        const gridConfig = { color: 'rgba(255, 255, 255, 0.05)', borderColor: 'rgba(255, 255, 255, 0.1)' };


        if(isSingleDie) {

            let labels = [];

            let data = [];

            let pointColors = [];

            let pointRadii = [];


            for(let x=1; x<=24; x++) {

                let testArr = new Array(x).fill(d1);

                let res = calcPkForPackage(testArr, red, grayCount, grayVal, steps);

                labels.push(x);

                data.push((res.pk * 100).toFixed(1));

                

                if(x === c1) {

                    pointColors.push('#00A8FF'); // Cyan highlight

                    pointRadii.push(6);

                } else {

                    pointColors.push('#374151');

                    pointRadii.push(3);

                }

            }


            myChart = new Chart(ctx, {

                type: 'line',

                data: {

                    labels: labels,

                    datasets: [{

                        label: 'Pk vs Total Salvoes',

                        data: data,

                        borderColor: '#00A8FF', // Cyan line

                        backgroundColor: 'rgba(0, 168, 255, 0.1)',

                        pointBackgroundColor: pointColors,

                        pointBorderColor: '#111827',

                        pointBorderWidth: 2,

                        pointRadius: pointRadii,

                        fill: true,

                        tension: 0.3

                    }]

                },

                options: {

                    responsive: true,

                    plugins: { title: { display: false }, legend: { display: false } },

                    scales: { 

                        y: { beginAtZero: true, max: 100, grid: gridConfig, title: { display: true, text: 'Probability (%)', color: '#6B7280' } }, 

                        x: { grid: gridConfig, title: { display: true, text: 'Number of Salvoes Allocated', color: '#6B7280' } } 

                    }

                }

            });

        } else {

            let labels = [];

            let data = [];

            let bgColors = [];

            let borderColors = [];


            for(let i=0; i<currentRes.pmf.length; i++) {

                labels.push(i + ' Hits');

                data.push((currentRes.pmf[i] * 100).toFixed(1));

                // Cyan for success, Dark Grey/Red tint for failure

                bgColors.push(i >= steps ? 'rgba(0, 168, 255, 0.8)' : 'rgba(229, 57, 53, 0.2)');

                borderColors.push(i >= steps ? '#00A8FF' : '#E53935');

            }


            myChart = new Chart(ctx, {

                type: 'bar',

                data: {

                    labels: labels,

                    datasets: [{

                        label: 'Probability of Exact Hits',

                        data: data,

                        backgroundColor: bgColors,

                        borderColor: borderColors,

                        borderWidth: 1,

                        borderRadius: 4

                    }]

                },

                options: {

                    responsive: true,

                    plugins: { title: { display: false }, legend: {display: false} },

                    scales: { 

                        y: { beginAtZero: true, grid: gridConfig, title: { display: true, text: 'Probability (%)', color: '#6B7280' } },

                        x: { grid: { display: false } }

                    }

                }

            });

        }

    }


    calculate();

</script>
